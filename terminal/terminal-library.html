terminal

terminal library

Provides the main interface of this package, Surface, and related types.

import 'package:dt/terminal.dart';

void main() {
  final terminal = Surface.fromStdio();
  terminal.draw((frame) {
    frame.draw((buffer) {
      buffer.print(0, 0, 'Hello, World!');
    });
  });
  terminal.dispose();
}
